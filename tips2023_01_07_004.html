    • from string import *
      from random import *

      def password_gen():
          L,U,D=ascii_lowercase, ascii_uppercase, digits
          pw = [choice(L),choice(U),choice(D)]
          pw += [choice(U+L+D)for i in range(randint(3, 17))]
          shuffle(pw)
          return "".join(pw)
