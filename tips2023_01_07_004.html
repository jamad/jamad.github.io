    • function passwordGen() {
        // Generate a random number between 6 and 20
        const check = Math.floor(Math.random() * (20 - 6 + 1) + 6);
        // Initialize the password as an empty string
        let password = "";
        // Generate a password until it is of the required length
        while (password.length < check) {
          // Choose a random lowercase letter
          password += String.fromCharCode(97 + Math.floor(Math.random() * 26));
          // Choose a random uppercase letter
          password += String.fromCharCode(65 + Math.floor(Math.random() * 26));
          // Choose a random number
          password += Math.floor(Math.random() * 10);
        }
        // Return the first `check` characters of the password
        return password.slice(0, check);
      }
