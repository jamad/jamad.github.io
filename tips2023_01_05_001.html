Q

    * 2023-01-04: (javascript css html) how to create the simple clock?

A

the following html file can show the clock like the following

   

"D:\myworks\project_github\practiceJavascript\practice_clock_javascript_css.html"

<style>
  #clock {
  text-align: center;
  font-size: 36px;
}
</style>

<div id="clock"></div>

<script>
  function pad(num) {
    return num.toString().padStart(2, '0');
  }

  function updateClock() {
    var now = new Date();
    var hours = pad(now.getHours());
    var minutes = pad(now.getMinutes());
    var seconds = pad(now.getSeconds());
    document.getElementById('clock').innerHTML = hours + ':' + minutes + ':' + seconds;
  }

  setInterval(updateClock, 1000);
</script>
