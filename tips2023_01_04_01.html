#import tkinter as tk
from tkinter import filedialog ,Tk,Frame,IntVar,Label,Radiobutton,Button,Listbox,END
from subprocess import run
from tkinterdnd2  import DND_FILES, TkinterDnD

root = TkinterDnD.Tk()  # NB - use this instead of tk.Tk()

mylistbox=Listbox(root,width=200,height=16)# width and height by character count

mylistbox.insert(1, "dragdrop file here")
mylistbox.drop_target_register(DND_FILES)

def mycallbackfunc(event):

    # this is very important to get the file path
    files = mylistbox.tk.splitlist(event.data) # ref - https://github.com/Eliav2/tkinterdnd2/blob/master/demos/demo_files_and_text.py

    #print(files)
    for filefullpath in files:
        import os
        filepath, basename = os.path.split(filefullpath)
        BASENAME = os.path.splitext(basename)
        print('base name : ',basename, BASENAME)
        print('started conversion ...',filefullpath)
        mylistbox.insert(END, 'started conversion ...')
        mylistbox.insert(END, filefullpath)

        # get movie height by cv2  # https://stackoverflow.com/questions/7348505/get-dimensions-of-a-video-file
        import cv2
        vid=cv2.VideoCapture(filefullpath)
        height=int(vid.get(cv2.CAP_PROP_FRAME_HEIGHT))
        if not height:
            print('error!')
            mylistbox.insert(END, 'error!!!!!!!!!!! '+event.data)
            continue

        height=min(height,480)
        print('movie height :',height)
        vid.release()# important to delete the file!

        new_filefullpath=f"{filepath}/{BASENAME[0]}_mini.mp4"
        print('new file path :',new_filefullpath)

        # ffmpeg conversion
        # automatic width divisible by 2 by height https://superuser.com/questions/624563/how-to-resize-a-video-to-make-it-smaller-with-ffmpeg
        mycommand=f'ffmpeg -n -i "{filefullpath}" -filter:v scale=-2:{height} -c:a copy "{new_filefullpath}"'# double quotation is critically important!
        os.system(mycommand)


        # move the original file to recycle bin
        from send2trash import send2trash # pip install Send2Trash
        try:send2trash(filefullpath.replace('/','\\')) #send2trash requires backslash
        except Exception as e:print('error',e)
       
        mylistbox.insert(END, 'done : '+event.data)

mylistbox.dnd_bind('<<Drop>>', mycallbackfunc)

mylistbox.pack()
root.mainloop()
