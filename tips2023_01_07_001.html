  •     2023-01-07: (python) how to compare 2 code solutions?

A

  •     下の二つのsolutionの結果を比較する
def solutionA(N,M,LINES):
    from collections import defaultdict
    D=defaultdict(int) # for []
   
    for L in LINES:
        u,v=map(int,L.split())
        if v<u:u,v=v,u
        D[v]=u
    #print(D)
    #print(sum(len(D[k]) for k in D))
   
   
    ROOT=set()
    for i in range(1,N+1):
        node=i
        while node in D:node=D[node]
        ROOT.add(node)
       
    return len(ROOT)		 def solutionB(N,M,LINES):
    D=defaultdict(list) # for []
    for L in LINES:
        u,v=map(int,L.split())
        D[u].append(v)
        D[v].append(u)
       
    print(D)
   
    V=[0]*(N+1)
    ans=0
   
    for i in range(1,N+1):
        print(V)
        if V[i]:continue
        ans+=1
        V[i]=1
       
        Q=[i]
        for c in Q:
            for node in D[c]:
                if V[node]:continue
                V[node]=1
                Q.append(node)
       
    return ans
