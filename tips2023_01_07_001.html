def solutionA(N,M,LINES):
    from collections import defaultdict
    D=defaultdict(int) # for []
   
    for L in LINES:
        u,v=map(int,L.split())
        if v<u:u,v=v,u
        D[v]=u
    #print(D)
    #print(sum(len(D[k]) for k in D))
   
   
    ROOT=set()
    for i in range(1,N+1):
        node=i
        while node in D:node=D[node]
        ROOT.add(node)
       
    return len(ROOT)
 def solutionB(N,M,LINES):
