Q
    (Blender) how to create the button in ImageEditor?
A
  •   the following can
    • import bpy

      # define a function that will be called when the button is pressed
      def my_button_callback(self, context):
          # put your desired logic here
          print("Button was pressed!")

      # create a new button
      class MY_OT_Button(bpy.types.Operator):
          bl_idname = "my.button"
          bl_label = "My Button"

          def execute(self, context):
              my_button_callback(self, context)
              return {'FINISHED'}

      # register the button
      bpy.utils.register_class(MY_OT_Button)

      # create a panel with a button
      class MY_PT_Panel(bpy.types.Panel):
          bl_label = "My Panel"
          bl_idname = "MY_PT_Panel"
          bl_space_type = 'IMAGE_EDITOR'
          bl_region_type = 'UI'

          def draw(self, context):
              layout = self.layout
              layout.operator("my.button")

      # register the panel
      bpy.utils.register_class(MY_PT_Panel)
